from threading import Thread, Event
from pynput import mouse
import pyautogui
import time

CPS = 16  # Clicks per second
INTERVAL = 1 / CPS

running = False
stop_event = Event()
mouse_down = False

def click_loop():
    global running
    while not stop_event.is_set():
        if running and mouse_down:
            pyautogui.click()
            time.sleep(INTERVAL)
        else:
            time.sleep(0.01)

def on_click(x, y, button, pressed):
    global mouse_down
    if button.name == "left":
        mouse_down = pressed

def toggle():
    global running
    running = not running
    status_label.config(text=f"Status: {'ON' if running else 'OFF'}")

# GUI
root = tk.Tk()
root.title("Auto Clicker")
root.geometry("250x100")

toggle_button = tk.Button(root, text="AÃ§/Kapat", command=toggle, width=15)
toggle_button.pack(pady=10)

status_label = tk.Label(root, text="Status: OFF")
status_label.pack()

# Mouse dinleyici
listener = mouse.Listener(on_click=on_click)
listener.start()

# Click thread
thread = Thread(target=click_loop, daemon=True)
thread.start()

def on_close():
    stop_event.set()
    listener.stop()
    root.destroy()

root.protocol("WM_DELETE_WINDOW", on_close)
root.mainloop()
